   /apps/fet/etc/tx/castor.conf
   --
   active yes
   log_roll_over  daily
   sleep_timer    10
   debug_level    3
   chmod          000
   destination ftp://am:Pr2namPW@castor/data/hoho/out WHATFN
   imask *:*:*:*:*
